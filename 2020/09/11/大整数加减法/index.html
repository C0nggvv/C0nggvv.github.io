大整数加减法C++代码

/**
 * 大整数加法（如99999999999999999+999999999999999)
 * 字符串用string存储
 * 参数s1,s2为得求和的两个大整数
 * 返回值为一个string类型字符串，其值为求和结果
 */
string add(string s1,string s2){
    int len1=s1.length();
    int len2=s2.length();
    string s="";
    int c=0;
    int i=len1-1,j=len2-1;
    while(i>=0 && j>=0){
        int cal=s1[i--]-'0'+s2[j--]-'0'+c;
        c=cal/10;
        s=char(cal%10+'0')+s;
    }
    while(i>=0){
        int cal=s1[i--]-'0'+c;
        c=cal/10;
        s=char(cal%10+'0')+s;
    }
    while(j>=0){
        int cal=s2[j--]-'0'+c;
        c=cal/10;
        s=char(cal%10+'0')+s;
    }
    if(c){
        s=char(c+'0')+s;
    }
    return s;
}

/**
 * 大整数乘一个一位数（如999999999999999*4）
 * 大整数、返回值用string,一位数用char存储
 * 参数str为大整数，p为一位数（char类型）
 * 返回值为乘积结果string类型
 */
string muti(string str,char p){
    string s="";
    int len=str.length();
    int c=0;
    int i=len-1;
    while(i>=0){
        int cal=(str[i--]-'0')*(p-'0')+c;
        c=cal/10;
        s=char(cal%10+'0')+s;
    }
    if(c){
        s=char(c+'0')+s;
    }
    return s;
}

/**
 * 大整数乘法(如99999999999999999*99999999999999999)
 * 参数为两个非负大整数，类型为string
 * 返回值为大整数乘积结果，类型为string
 * 注：需要调用上面的muti(string,string)函数
 */
string mutiple(string s1,string s2){
    if(s1=="0" || s2=="0"){
        return "0";
    }
    int len=s2.length();
    string ans="";
    string str="";
    for(int i=len-1;i>=0;i--){
        ans=add(ans,muti(s1,s2[i])+str);
        str+='0';
    }
    return ans;
}

/**
 * 整数转字符串
 * 参数为一非负整数，可以修改为long long
 * 返回值为该整数对应的字符串
 * 注：手打没有运行过，应该没问题
 */
string change(int num){
    if(num==0)
        return "0";
    string s="";
    while(num){
        s=char(num%10+'0')+s;
        num/=10;
    }
    return s;
}
