int gcd(int a,int b){
    return b==0?a:gcd(b,a%b);
}

//拓展欧几里得算法
int ex_gcd(int a,int b,int& x,int& y) 
{
    int d=a;
    if(b!=0){
        d = ex_gcd(b,a%b,y,x);
        y-=(a/b)*x;
    }else{
        x=1;y=0;
    }
    return d;
}

//素性判定
bool Is_prime(int n){
    for(int i=2;i*i<=n;i++){
        if(n%i==0)
            return false;
    }
    return n!=1;
}

//约数枚举
vector<int> divisor(int n){
    vector<int> res;
    for(int i=1;i*i<=n;i++){
        if(n%i==0){
            res.push_back(i);
            if(i!=n/i)
                res.push_back(n/i);
        }
    }
    return res;
}

//整数分解
map<int,int> prime_factor(int n){
    map<int,int> res;
    for(int i=2;i*i<=n;i++){
        while(n%i==0){
            res[i]++;
            n/=i;
        }
    }
    if(n!=1)
        res[n]=1;
    return res;
}

//埃氏筛法
int sieve(int n){
    int p=0;
    for(int i=0;i<=n;i++)
        is_prime[i]=true;
    is_prime[0]=is_prime[1]=false;
    for(int i=2;i<=n;i++){
        if(is_prime[i]){
            prime[p++]=i;
            for(int j=2*i;j<=n;j+=i){
                is_prime[j]=false;
            }
        }
    }
    return p;
}

//求区间素数
void segment_sieve(ll a,ll b){
    for(int i=0;(ll)i*i<b;i++)
        is_prime_small[i]=true;
    for(int i=0;i<b-a;i++)
        is_prime[i]=true;
    for(int i=2;(ll)i*i<b;i++){
        if(is_prime_small[i]){
            for(int j=2*i;(ll)j*j<=b;j+=i)
                is_prime_small[j]=false;
            for(ll j=max(2LL,((a+i-1)/i)*i);j<b;j+=i)
                is_prime[j-a]=false;
        }
    }
}

//快速幂
ll mod_pow(ll x,ll n,ll mod){
    ll res=1;
    while(n){
        if(n&1)
            res=res*x%mod;
        x=x*x%mod; 
        n>>=1;
    }
    return res;
}

//快速乘
ll quick_mul(ll a,ll b,ll mod){
    ll res=0;
    while(b){
        if(b%1)
            res=(res+a)%mod;
        a=(a+a)%mod;
        b>>=1; 
    }
    return res;
}
