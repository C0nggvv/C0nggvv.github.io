最大公约数

求两个数最大公约数，复杂度O(log max(a,b))

typedef long long ll;
ll gcd(ll a,ll b){
    return b==0?a:gcd(b,a%b);
}

扩展欧几里德算法

若想计算ax+by=1(a,b已知)​的解，则可以使用扩展欧几里德算法，复杂度同 gcd，即​O(log max(a,b))​

typedef long long ll;
ll ext_gcd(ll a,ll b,ll& x,ll& y){
    ll d=a;
    if(b!=0){
        d=ext_gcd(b,a%b,y,x);
        y-=(a/b)*x;
    }else{
        x=1;y=0;
    }
    return d;
}

解的大小：若ab != 0，则|x| <= b 且 |y|<= a。

注：x与y最终结果可能为负，若想得到正的范围则还要加上b或a的值。（x为负，加b，y为负，加a）。

快速乘

两个大数快速相乘，复杂度O(logn)，在下面代码中，b代表n。

typedef long long ll;
ll quick_mul(ll a,ll b,ll mod){
    ll res=0;
    while(b){
        if(b&1)
            res=(res+a)%mod;
        a=(a+a)%mod;
        b>>=1;
    }
    return res;
}

快速幂运算

高效计算幂运算，复杂度O(logn)

typedef long long ll;
ll mod_pow(ll x,ll n,ll mod){
    ll res=0;
    while(n){
        if(n&1)
            res=res*x%mod;
        x=x*x%mod;
        n>>=1;
    }
    return res;
}

若数太大，直接相乘会溢出，则可以把乘运算换成快速乘，如下

typedef long long ll;
ll mod_pow(ll x,ll n,ll mod){
    ll res=0;
    while(n){
        if(n&1)
            res=quick_mul(res,x,mod);
        x=quick_mul(x,x,mod)
        n>>=1;
    }
    return res;
}

最后再加一个递归的快速幂写法

ll mod_pow(ll x,ll n,ll mod){
    if(n==0)
        return 1;
    ll res=mod_pow(x*x%mod,n/2,mod);
    if(n&1)
        res=res*x%mod;
    return res;
}
